# configuration file /etc/nginx/sites-enabled/aienter.in:
server {
    server_name aienter.in www.aienter.in;

    location /.well-known/acme-challenge/ {
        alias /var/www/ai-enter/.well-known/acme-challenge/;
        try_files $uri =404;
        allow all;
    }

    location / {
        proxy_pass http://127.0.0.1:3040;
        proxy_http_version 1.1;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_cache_bypass $http_upgrade;
    }

    listen [::]:443 ssl ipv6only=on; # managed by Certbot
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/aienter.in/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/aienter.in/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}

server {
    if ($host = www.aienter.in) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    if ($host = aienter.in) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    listen 80;
    listen [::]:80;
    server_name aienter.in www.aienter.in;
    return 404; # managed by Certbot
}
